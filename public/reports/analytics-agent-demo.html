Methodology

Data Sources: olist_customers_dataset, olist_geolocation_dataset, olist_order_items_dataset, olist_order_payments_dataset, olist_order_reviews_dataset, olist_orders_dataset, olist_products_dataset, olist_sellers_dataset, product_category_name_translation

Analysis Approach: To answer what product categories drive the most revenue and how this has changed over the past 12 months, we need to: (1) compute total revenue by category across all time to establish overall rankings, (2) break down category revenue by month over the past 12 months to reveal trends and shifts. Revenue is measured using order item prices (sum of price) joined with product category data and filtered to delivered/valid orders. We use order_purchase_timestamp as the time dimension. Category names are translated to English via the translation table. We avoid double-counting by using item-level price rather than payment_value (which can span multiple items).

SQL Queries Used

Rank all product categories by total revenue (sum of item prices) to identify the top revenue-driving categories overall, providing a baseline for comparison.

SELECT
    COALESCE(t.product_category_name_english, p.product_category_name, 'Unknown') AS product_category,
    SUM(oi.price) AS total_revenue,
    COUNT(DISTINCT oi.order_id) AS total_orders,
    COUNT(oi.order_item_id) AS total_items_sold
FROM olist_order_items_dataset oi
JOIN olist_orders_dataset o ON oi.order_id = o.order_id
JOIN olist_products_dataset p ON oi.product_id = p.product_id
LEFT JOIN product_category_name_translation t ON p.product_category_name = t.product_category_name
WHERE o.order_status = 'delivered'
GROUP BY COALESCE(t.product_category_name_english, p.product_category_name, 'Unknown')
ORDER BY total_revenue DESC

Show monthly revenue by product category over the past 12 months to reveal how each category's contribution has changed over time, identifying growth or decline trends.

WITH max_date AS (
    SELECT MAX(order_purchase_timestamp) AS max_ts
    FROM olist_orders_dataset
),
filtered_orders AS (
    SELECT
        o.order_id,
        o.order_purchase_timestamp
    FROM olist_orders_dataset o, max_date
    WHERE o.order_status = 'delivered'
      AND o.order_purchase_timestamp >= max_date.max_ts - INTERVAL '12 months'
),
joined AS (
    SELECT
        DATE_TRUNC('month', fo.order_purchase_timestamp) AS order_month,
        COALESCE(t.product_category_name_english, p.product_category_name, 'unknown') AS category,
        SUM(oi.price) AS monthly_revenue
    FROM filtered_orders fo
    JOIN olist_order_items_dataset oi ON fo.order_id = oi.order_id
    JOIN olist_products_dataset p ON oi.product_id = p.product_id
    LEFT JOIN product_category_name_translation t ON p.product_category_name = t.product_category_name
    GROUP BY 1, 2
)
SELECT
    order_month,
    category,
    monthly_revenue
FROM joined
ORDER BY order_month ASC, monthly_revenue DESC

Compare each category's share of total revenue in the most recent 6 months versus the prior 6 months to highlight which categories are gaining or losing revenue share.

WITH max_date AS (
    SELECT MAX(order_purchase_timestamp) AS max_ts
    FROM olist_orders_dataset
),
period_bounds AS (
    SELECT
        max_ts,
        max_ts - INTERVAL '6 months' AS recent_start,
        max_ts - INTERVAL '12 months' AS prior_start
    FROM max_date
),
filtered_orders AS (
    SELECT
        o.order_id,
        o.order_purchase_timestamp,
        CASE
            WHEN o.order_purchase_timestamp >= pb.recent_start THEN 'recent_6_months'
            WHEN o.order_purchase_timestamp >= pb.prior_start THEN 'prior_6_months'
        END AS period
    FROM olist_orders_dataset o
    CROSS JOIN period_bounds pb
    WHERE o.order_status = 'delivered'
      AND o.order_purchase_timestamp >= pb.prior_start
),
joined AS (
    SELECT
        fo.period,
        COALESCE(t.product_category_name_english, p.product_category_name, 'unknown') AS category,
        oi.price
    FROM filtered_orders fo
    JOIN olist_order_items_dataset oi ON fo.order_id = oi.order_id
    JOIN olist_products_dataset p ON oi.product_id = p.product_id
    LEFT JOIN product_category_name_translation t ON p.product_category_name = t.product_category_name
    WHERE fo.period IS NOT NULL
),
category_period_revenue AS (
    SELECT
        category,
        period,
        SUM(price) AS revenue
    FROM joined
    GROUP BY category, period
),
period_totals AS (
    SELECT
        period,
        SUM(revenue) AS total_revenue
    FROM category_period_revenue
    GROUP BY period
),
shares AS (
    SELECT
        cpr.category,
        cpr.period,
        cpr.revenue,
        pt.total_revenue,
        ROUND(100.0 * cpr.revenue / pt.total_revenue, 4) AS revenue_share_pct
    FROM category_period_revenue cpr
    JOIN period_totals pt ON cpr.period = pt.period
),
pivoted AS (
    SELECT
        category,
        MAX(CASE WHEN period = 'recent_6_months' THEN revenue END) AS recent_revenue,
        MAX(CASE WHEN period = 'prior_6_months' THEN revenue END) AS prior_revenue,
        MAX(CASE WHEN period = 'recent_6_months' THEN revenue_share_pct END) AS recent_share_pct,
        MAX(CASE WHEN period = 'prior_6_months' THEN revenue_share_pct END) AS prior_share_pct
    FROM shares
    GROUP BY category
)
SELECT
    category,
    ROUND(COALESCE(recent_revenue, 0), 2) AS recent_6m_revenue,
    ROUND(COALESCE(prior_revenue, 0), 2) AS prior_6m_revenue,
    ROUND(COALESCE(recent_share_pct, 0), 4) AS recent_6m_share_pct,
    ROUND(COALESCE(prior_share_pct, 0), 4) AS prior_6m_share_pct,
    ROUND(COALESCE(recent_share_pct, 0) - COALESCE(prior_share_pct, 0), 4) AS share_change_pct_points
FROM pivoted
ORDER BY share_change_pct_points DESC

Provide monthly order count and average order value by category over the past 12 months for the top categories, to distinguish whether revenue changes are driven by volume or pricing shifts.

WITH date_range AS (
    SELECT
        MAX(order_purchase_timestamp) AS max_date,
        MAX(order_purchase_timestamp) - INTERVAL '12 months' AS min_date
    FROM olist_orders_dataset
),
top_categories AS (
    SELECT
        COALESCE(t.product_category_name_english, p.product_category_name) AS category,
        SUM(oi.price) AS total_revenue
    FROM olist_order_items_dataset oi
    JOIN olist_orders_dataset o ON oi.order_id = o.order_id
    JOIN olist_products_dataset p ON oi.product_id = p.product_id
    LEFT JOIN product_category_name_translation t ON p.product_category_name = t.product_category_name
    JOIN date_range dr ON o.order_purchase_timestamp >= dr.min_date
        AND o.order_purchase_timestamp <= dr.max_date
    WHERE o.order_status NOT IN ('canceled', 'unavailable')
      AND p.product_category_name IS NOT NULL
    GROUP BY 1
    ORDER BY total_revenue DESC
    LIMIT 10
),
monthly_data AS (
    SELECT
        DATE_TRUNC('month', o.order_purchase_timestamp) AS order_month,
        COALESCE(t.product_category_name_english, p.product_category_name) AS category,
        COUNT(DISTINCT oi.order_id) AS order_count,
        SUM(oi.price) AS total_revenue,
        SUM(oi.price) / COUNT(DISTINCT oi.order_id) AS avg_order_value
    FROM olist_order_items_dataset oi
    JOIN olist_orders_dataset o ON oi.order_id = o.order_id
    JOIN olist_products_dataset p ON oi.product_id = p.product_id
    LEFT JOIN product_category_name_translation t ON p.product_category_name = t.product_category_name
    JOIN date_range dr ON o.order_purchase_timestamp >= dr.min_date
        AND o.order_purchase_timestamp <= dr.max_date
    WHERE o.order_status NOT IN ('canceled', 'unavailable')
      AND p.product_category_name IS NOT NULL
    GROUP BY 1, 2
)
SELECT
    md.order_month,
    md.category,
    md.order_count,
    ROUND(md.total_revenue, 2) AS total_revenue,
    ROUND(md.avg_order_value, 2) AS avg_order_value
FROM monthly_data md
INNER JOIN top_categories tc ON md.category = tc.category
ORDER BY md.order_month ASC, tc.total_revenue DESC
