Methodology

Data Sources: campaigns, pageviews, sessions, transactions

Analysis Approach: To answer the ROI and campaign scaling question, we need to: (1) compute per-campaign spend, revenue, and ROI by joining campaigns → sessions → transactions; (2) analyze engagement and conversion metrics per campaign to understand quality of traffic; (3) break down performance by channel to see if certain channels drive better returns. We must account for the 57.7% null campaign_id in sessions by restricting to attributed sessions only. ROI is defined as (Revenue - Spend) / Spend * 100.

SQL Queries Used

Calculate total spend, total revenue, ROI, number of attributed sessions, number of transactions, and conversion rate for each campaign. This is the primary metric for deciding which campaigns to scale or cut.

SELECT
  c.campaign_id,
  c.campaign_name,
  c.channel,
  c.total_spend,
  COUNT(DISTINCT s.session_id) AS attributed_sessions,
  COUNT(DISTINCT t.transaction_id) AS num_transactions,
  SUM(t.revenue) AS total_revenue,
  ROUND(
    (SUM(t.revenue) - c.total_spend) / NULLIF(c.total_spend, 0) * 100,
    2
  ) AS roi_pct,
  ROUND(
    COUNT(DISTINCT t.transaction_id) * 1.0 / NULLIF(COUNT(DISTINCT s.session_id), 0) * 100,
    2
  ) AS conversion_rate_pct
FROM campaigns c
LEFT JOIN sessions s
  ON s.campaign_id = c.campaign_id
LEFT JOIN transactions t
  ON t.session_id = s.session_id
GROUP BY
  c.campaign_id,
  c.campaign_name,
  c.channel,
  c.total_spend
ORDER BY total_revenue DESC

Evaluate session quality per campaign by computing average pages viewed, average session duration, bounce rate, and average revenue per transaction. This helps distinguish campaigns with high spend but low engagement or poor conversion quality.

SELECT
    c.campaign_id,
    c.campaign_name,
    COUNT(DISTINCT s.session_id) AS total_sessions,
    ROUND(AVG(s.pages_viewed), 2) AS avg_pages_viewed,
    ROUND(AVG(s.duration_seconds), 2) AS avg_session_duration_seconds,
    ROUND(100.0 * SUM(s.bounced) / COUNT(s.session_id), 2) AS bounce_rate_pct,
    COUNT(DISTINCT t.transaction_id) AS total_transactions,
    ROUND(SUM(t.revenue) / NULLIF(COUNT(DISTINCT t.transaction_id), 0), 2) AS avg_revenue_per_transaction
FROM campaigns c
LEFT JOIN sessions s
    ON s.campaign_id = c.campaign_id
LEFT JOIN transactions t
    ON t.session_id = s.session_id
GROUP BY
    c.campaign_id,
    c.campaign_name
ORDER BY
    avg_revenue_per_transaction DESC

Aggregate ROI, revenue, spend, sessions, and conversions by marketing channel (paid_search, social, etc.) to identify whether performance differences are campaign-specific or channel-structural, informing scaling strategy at the channel level.

WITH channel_spend AS (
    SELECT
        channel,
        SUM(total_spend) AS total_spend
    FROM campaigns
    GROUP BY channel
),
channel_sessions AS (
    SELECT
        s.channel,
        COUNT(DISTINCT s.session_id) AS total_sessions
    FROM sessions s
    GROUP BY s.channel
),
channel_revenue AS (
    SELECT
        s.channel,
        SUM(t.revenue) AS total_revenue,
        COUNT(DISTINCT t.transaction_id) AS total_conversions
    FROM sessions s
    INNER JOIN transactions t ON s.session_id = t.session_id
    GROUP BY s.channel
)
SELECT
    css.channel,
    COALESCE(cr.total_revenue, 0) AS total_revenue,
    COALESCE(csp.total_spend, 0) AS total_spend,
    COALESCE(css.total_sessions, 0) AS total_sessions,
    COALESCE(cr.total_conversions, 0) AS total_conversions,
    CASE
        WHEN COALESCE(csp.total_spend, 0) = 0 THEN NULL
        ELSE ROUND((COALESCE(cr.total_revenue, 0) - csp.total_spend) / csp.total_spend, 4)
    END AS roi,
    CASE
        WHEN COALESCE(csp.total_spend, 0) = 0 THEN NULL
        ELSE ROUND(COALESCE(cr.total_revenue, 0) / csp.total_spend, 4)
    END AS roas,
    CASE
        WHEN COALESCE(css.total_sessions, 0) = 0 THEN NULL
        ELSE ROUND(CAST(COALESCE(cr.total_conversions, 0) AS DOUBLE) / css.total_sessions * 100, 4)
    END AS conversion_rate_pct
FROM channel_sessions css
LEFT JOIN channel_spend csp ON css.channel = csp.channel
LEFT JOIN channel_revenue cr ON css.channel = cr.channel
ORDER BY total_revenue DESC

Track weekly revenue and session volume attributed to each campaign across its active period to identify trends, ramp-up patterns, or decay in performance — useful for deciding whether underperformers are early-stage or genuinely declining.

WITH campaign_weeks AS (
    SELECT
        c.campaign_id,
        c.campaign_name,
        c.channel,
        c.start_date,
        c.end_date,
        -- Generate weekly buckets within the campaign's active period
        DATE_TRUNC('week', gs.week_start) AS week_start
    FROM campaigns c
    CROSS JOIN (
        SELECT UNNEST(GENERATE_SERIES(
            (SELECT MIN(start_date) FROM campaigns),
            (SELECT MAX(end_date) FROM campaigns),
            INTERVAL '7 days'
        )) AS week_start
    ) gs
    WHERE gs.week_start <= c.end_date
      AND gs.week_start + INTERVAL '6 days' >= c.start_date
),
session_revenue AS (
    SELECT
        s.session_id,
        s.campaign_id,
        s.session_date,
        COALESCE(t.revenue, 0) AS revenue
    FROM sessions s
    LEFT JOIN transactions t ON s.session_id = t.session_id
    WHERE s.campaign_id IS NOT NULL
),
weekly_metrics AS (
    SELECT
        cw.campaign_id,
        cw.campaign_name,
        cw.channel,
        cw.start_date AS campaign_start,
        cw.end_date AS campaign_end,
        cw.week_start,
        cw.week_start + INTERVAL '6 days' AS week_end,
        COUNT(DISTINCT sr.session_id) AS session_count,
        SUM(sr.revenue) AS total_revenue,
        ROW_NUMBER() OVER (
            PARTITION BY cw.campaign_id
            ORDER BY cw.week_start
        ) AS week_number
    FROM campaign_weeks cw
    LEFT JOIN session_revenue sr
        ON cw.campaign_id = sr.campaign_id
        AND sr.session_date >= cw.week_start
        AND sr.session_date <= CAST(cw.week_start + INTERVAL '6 days' AS DATE)
    GROUP BY
        cw.campaign_id,
        cw.campaign_name,
        cw.channel,
        cw.start_date,
        cw.end_date,
        cw.week_start
)
SELECT
    campaign_id,
    campaign_name,
    channel,
    campaign_start,
    campaign_end,
    week_start,
    CAST(week_end AS DATE) AS week_end,
    week_number,
    session_count,
    COALESCE(total_revenue, 0) AS total_revenue
FROM weekly_metrics
ORDER BY campaign_id, week_start
