Methodology

Data Sources: daily_prices, tickers

Analysis Approach: To calculate the Sharpe ratio for each stock over the last 5 years, we need to: (1) compute daily log or simple returns from closing prices, (2) calculate annualized mean return and annualized volatility (std dev) per ticker, (3) derive the Sharpe ratio as (annualized_mean_return - risk_free_rate) / annualized_volatility — using 0 or a constant risk-free rate proxy since no rate table is available, (4) join with tickers to get sector, and (5) rank to extract top 10 and bottom 10. We'll split this into a daily returns computation, a per-ticker Sharpe ratio summary with ranking, and a final filtered top/bottom 10 result.

Compute daily simple returns for each ticker over the last 5 years by using LAG on close prices, filtering to the relevant date window and excluding the benchmark ticker if present.

WITH max_date AS (
    SELECT MAX(date) AS max_dt FROM daily_prices
),
benchmark_tickers AS (
    -- Identify tickers not in the tickers table (likely benchmarks like SPY, QQQ, etc.)
    SELECT DISTINCT dp.ticker
    FROM daily_prices dp
    LEFT JOIN tickers t ON dp.ticker = t.ticker
    WHERE t.ticker IS NULL
),
priced AS (
    SELECT
        dp.date,
        dp.ticker,
        dp.close,
        t.sector,
        LAG(dp.close) OVER (PARTITION BY dp.ticker ORDER BY dp.date) AS prev_close
    FROM daily_prices dp
    INNER JOIN tickers t ON dp.ticker = t.ticker  -- excludes benchmark tickers not in tickers table
    CROSS JOIN max_date md
    WHERE dp.date >= (md.max_dt - INTERVAL '5 years')
      AND dp.ticker NOT IN (SELECT ticker FROM benchmark_tickers)
)
SELECT
    date,
    ticker,
    sector,
    close,
    prev_close,
    CASE
        WHEN prev_close IS NOT NULL AND prev_close <> 0
        THEN (close - prev_close) / prev_close
        ELSE NULL
    END AS daily_simple_return
FROM priced
WHERE prev_close IS NOT NULL
  AND prev_close <> 0
ORDER BY ticker, date

Aggregate daily returns per ticker to compute annualized mean return and annualized volatility, then calculate the Sharpe ratio (annualized_mean / annualized_stddev, assuming a 0% risk-free rate) for each stock, joined with sector from the tickers table.

WITH daily_returns AS (
    SELECT
        ticker,
        date,
        (close / LAG(close) OVER (PARTITION BY ticker ORDER BY date) - 1) AS daily_return
    FROM daily_prices
),
returns_filtered AS (
    SELECT
        ticker,
        daily_return
    FROM daily_returns
    WHERE daily_return IS NOT NULL
),
stats AS (
    SELECT
        ticker,
        AVG(daily_return) * 252 AS annualized_mean,
        STDDEV_SAMP(daily_return) * SQRT(252) AS annualized_volatility,
        COUNT(*) AS trading_days
    FROM returns_filtered
    GROUP BY ticker
)
SELECT
    s.ticker,
    t.sector,
    ROUND(s.annualized_mean * 100, 4) AS annualized_mean_pct,
    ROUND(s.annualized_volatility * 100, 4) AS annualized_volatility_pct,
    ROUND(
        CASE
            WHEN s.annualized_volatility = 0 OR s.annualized_volatility IS NULL THEN NULL
            ELSE s.annualized_mean / s.annualized_volatility
        END,
        4
    ) AS sharpe_ratio,
    s.trading_days
FROM stats s
JOIN tickers t ON s.ticker = t.ticker
ORDER BY sharpe_ratio DESC

WITH daily_returns AS (
    SELECT
        dp.ticker,
        t.sector,
        dp.date,
        dp.close,
        (dp.close - LAG(dp.close) OVER (PARTITION BY dp.ticker ORDER BY dp.date)) / LAG(dp.close) OVER (PARTITION BY dp.ticker ORDER BY dp.date) AS daily_return
    FROM daily_prices dp
    JOIN tickers t ON dp.ticker = t.ticker
),
stats AS (
    SELECT
        ticker,
        sector,
        AVG(daily_return) * 252 AS annualized_return,
        STDDEV(daily_return) * SQRT(252) AS annualized_volatility,
        COUNT(daily_return) AS n_obs
    FROM daily_returns
    WHERE daily_return IS NOT NULL
    GROUP BY ticker, sector
)
SELECT
    ticker,
    sector,
    ROUND(annualized_return * 100, 2) AS annualized_return_pct,
    ROUND(annualized_volatility * 100, 2) AS annualized_volatility_pct,
    ROUND(annualized_return / NULLIF(annualized_volatility, 0), 4) AS sharpe_ratio
FROM stats
WHERE annualized_volatility > 0
ORDER BY sharpe_ratio DESC
LIMIT 10
