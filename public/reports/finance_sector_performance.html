Calculate the S&P 500 benchmark annual return for 2024 to establish the performance threshold for comparison. Uses the 'Benchmark' sector ticker to get start-of-year and end-of-year close prices.

WITH benchmark_prices AS (
    SELECT
        dp.date,
        dp.close
    FROM daily_prices dp
    JOIN tickers t ON dp.ticker = t.ticker
    WHERE t.sector = 'Benchmark'
      AND YEAR(dp.date) = 2024
),
start_price AS (
    SELECT close AS start_close
    FROM benchmark_prices
    ORDER BY date ASC
    LIMIT 1
),
end_price AS (
    SELECT close AS end_close
    FROM benchmark_prices
    ORDER BY date DESC
    LIMIT 1
)
SELECT
    2024 AS year,
    s.start_close AS start_of_year_close,
    e.end_close AS end_of_year_close,
    ROUND(((e.end_close - s.start_close) / s.start_close) * 100, 2) AS annual_return_pct
FROM start_price s
CROSS JOIN end_price e

Calculate the average annual return for each sector in 2024 by computing each constituent stock's return (first to last close within 2024) and averaging by sector. Compare each sector's return to the S&P 500 benchmark to identify outperforming sectors and rank them.

WITH date_range AS (
    SELECT
        MIN(date) AS min_date,
        MAX(date) AS max_date
    FROM daily_prices
    WHERE date >= '2024-01-01' AND date < '2025-01-01'
),
first_close AS (
    SELECT
        dp.ticker,
        dp.close AS first_close
    FROM daily_prices dp
    INNER JOIN (
        SELECT ticker, MIN(date) AS first_date
        FROM daily_prices
        WHERE date >= '2024-01-01' AND date < '2025-01-01'
        GROUP BY ticker
    ) fd ON dp.ticker = fd.ticker AND dp.date = fd.first_date
),
last_close AS (
    SELECT
        dp.ticker,
        dp.close AS last_close
    FROM daily_prices dp
    INNER JOIN (
        SELECT ticker, MAX(date) AS last_date
        FROM daily_prices
        WHERE date >= '2024-01-01' AND date < '2025-01-01'
        GROUP BY ticker
    ) ld ON dp.ticker = ld.ticker AND dp.date = ld.last_date
),
stock_returns AS (
    SELECT
        f.ticker,
        (l.last_close - f.first_close) / f.first_close * 100 AS annual_return
    FROM first_close f
    JOIN last_close l ON f.ticker = l.ticker
),
sp500_return AS (
    SELECT
        AVG(annual_return) AS benchmark_return
    FROM stock_returns
),
sector_returns AS (
    SELECT
        t.sector,
        AVG(sr.annual_return) AS avg_sector_return,
        COUNT(DISTINCT sr.ticker) AS num_stocks
    FROM stock_returns sr
    JOIN tickers t ON sr.ticker = t.ticker
    GROUP BY t.sector
)
SELECT
    s.sector,
    ROUND(s.avg_sector_return, 2) AS avg_annual_return_pct,
    ROUND(sp.benchmark_return, 2) AS sp500_benchmark_return_pct,
    ROUND(s.avg_sector_return - sp.benchmark_return, 2) AS excess_return_vs_benchmark,
    CASE WHEN s.avg_sector_return > sp.benchmark_return THEN 'Outperforming' ELSE 'Underperforming' END AS vs_benchmark,
    s.num_stocks
FROM sector_returns s
CROSS JOIN sp500_return sp
ORDER BY avg_annual_return_pct DESC

Break down returns within outperforming sectors by industry to identify what drove sector outperformance. Computes average annual return at the industry level within each sector, helping pinpoint which sub-industries contributed most.

WITH date_range AS (
    SELECT MIN(date) AS min_date, MAX(date) AS max_date
    FROM daily_prices
),
ticker_returns AS (
    SELECT
        dp_start.ticker,
        (dp_end.close - dp_start.close) / NULLIF(dp_start.close, 0) AS total_return,
        DATEDIFF('day', dr.min_date, dr.max_date) / 365.0 AS years
    FROM daily_prices dp_start
    JOIN daily_prices dp_end
        ON dp_start.ticker = dp_end.ticker
    CROSS JOIN date_range dr
    WHERE dp_start.date = dr.min_date
      AND dp_end.date = dr.max_date
),
ticker_annual_returns AS (
    SELECT
        t.ticker,
        t.sector,
        t.industry,
        tr.total_return / NULLIF(tr.years, 0) AS annual_return
    FROM ticker_returns tr
    JOIN tickers t ON tr.ticker = t.ticker
),
sector_avg AS (
    SELECT
        sector,
        AVG(annual_return) AS sector_avg_return
    FROM ticker_annual_returns
    GROUP BY sector
),
overall_avg AS (
    SELECT AVG(annual_return) AS overall_avg_return
    FROM ticker_annual_returns
),
outperforming_sectors AS (
    SELECT s.sector, s.sector_avg_return
    FROM sector_avg s
    CROSS JOIN overall_avg o
    WHERE s.sector_avg_return > o.overall_avg_return
)
SELECT
    tar.sector,
    tar.industry,
    AVG(tar.annual_return) AS avg_annual_return,
    os.sector_avg_return,
    COUNT(tar.ticker) AS ticker_count
FROM ticker_annual_returns tar
JOIN outperforming_sectors os ON tar.sector = os.sector
GROUP BY tar.sector, tar.industry, os.sector_avg_return
ORDER BY avg_annual_return DESC

Identify the top 10 and bottom 10 individual stocks by annual return in 2024. Computes each stock's return as (year-end close - year-start close) / year-start close, includes sector and industry for context, and ranks stocks to surface the best and worst performers.

WITH year_prices AS (
    SELECT
        ticker,
        FIRST_VALUE(close) OVER (PARTITION BY ticker ORDER BY date ASC) AS year_start_close,
        LAST_VALUE(close) OVER (PARTITION BY ticker ORDER BY date ASC ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS year_end_close
    FROM daily_prices
    WHERE date >= '2024-01-01' AND date < '2025-01-01'
),
unique_prices AS (
    SELECT DISTINCT
        ticker,
        year_start_close,
        year_end_close
    FROM year_prices
),
returns AS (
    SELECT
        up.ticker,
        t.sector,
        t.industry,
        up.year_start_close,
        up.year_end_close,
        (up.year_end_close - up.year_start_close) / up.year_start_close AS annual_return
    FROM unique_prices up
    JOIN tickers t ON up.ticker = t.ticker
    WHERE up.year_start_close IS NOT NULL AND up.year_start_close != 0
),
top_10 AS (
    SELECT *, 'Top 10' AS rank_group, ROW_NUMBER() OVER (ORDER BY annual_return DESC) AS rank
    FROM returns
    ORDER BY annual_return DESC
    LIMIT 10
),
bottom_10 AS (
    SELECT *, 'Bottom 10' AS rank_group, ROW_NUMBER() OVER (ORDER BY annual_return ASC) AS rank
    FROM returns
    ORDER BY annual_return ASC
    LIMIT 10
)
SELECT
    rank_group,
    rank,
    ticker,
    sector,
    industry,
    ROUND(year_start_close, 2) AS year_start_close,
    ROUND(year_end_close, 2) AS year_end_close,
    ROUND(annual_return * 100, 2) AS annual_return_pct
FROM (
    SELECT * FROM top_10
    UNION ALL
    SELECT * FROM bottom_10
) combined
ORDER BY rank_group DESC, annual_return DESC
