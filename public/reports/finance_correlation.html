Methodology

Data Sources: daily_prices, tickers

Analysis Approach: To build a correlation matrix and identify the highest/lowest correlated pairs, we need daily returns for each non-SPY stock. We will: (1) compute daily log or simple returns per ticker from daily_prices, excluding SPY; (2) pivot the returns into a wide format and compute pairwise Pearson correlations across all stock pairs; (3) extract and rank all unique pairs to find the 10 lowest and 10 highest correlation pairs for the bar charts.

SQL Queries Used

Compute daily simple returns (close-to-close) for each ticker excluding SPY, to serve as the input for all correlation analyses.

SELECT
    dp.date,
    dp.ticker,
    dp.close,
    t.sector,
    (dp.close - LAG(dp.close) OVER (PARTITION BY dp.ticker ORDER BY dp.date)) 
        / LAG(dp.close) OVER (PARTITION BY dp.ticker ORDER BY dp.date) AS daily_return
FROM daily_prices dp
JOIN tickers t ON dp.ticker = t.ticker
WHERE dp.ticker <> 'SPY'
QUALIFY LAG(dp.close) OVER (PARTITION BY dp.ticker ORDER BY dp.date) IS NOT NULL
ORDER BY dp.ticker, dp.date

Compute the full pairwise Pearson correlation matrix of daily returns across all individual stocks (excluding SPY), to be rendered as a heatmap.

WITH daily_returns AS (
    SELECT
        date,
        ticker,
        (close - LAG(close) OVER (PARTITION BY ticker ORDER BY date)) / LAG(close) OVER (PARTITION BY ticker ORDER BY date) AS daily_return
    FROM daily_prices
    WHERE ticker != 'SPY'
),
filtered_returns AS (
    SELECT date, ticker, daily_return
    FROM daily_returns
    WHERE daily_return IS NOT NULL
)
SELECT
    a.ticker AS ticker_a,
    b.ticker AS ticker_b,
    CORR(a.daily_return, b.daily_return) AS correlation
FROM filtered_returns a
JOIN filtered_returns b
    ON a.date = b.date
    AND a.ticker <= b.ticker
GROUP BY a.ticker, b.ticker
ORDER BY a.ticker, b.ticker

Identify the 10 stock pairs with the lowest pairwise Pearson correlation of daily returns (excluding SPY), for display as a horizontal bar chart labeled 'TICKER1 / TICKER2'.

WITH daily_returns AS (
    SELECT
        date,
        ticker,
        (close - LAG(close) OVER (PARTITION BY ticker ORDER BY date)) / LAG(close) OVER (PARTITION BY ticker ORDER BY date) AS daily_return
    FROM daily_prices
    WHERE ticker != 'SPY'
),
returns_filtered AS (
    SELECT date, ticker, daily_return
    FROM daily_returns
    WHERE daily_return IS NOT NULL
),
ticker_list AS (
    SELECT DISTINCT ticker FROM returns_filtered
),
pairs AS (
    SELECT
        a.ticker AS ticker1,
        b.ticker AS ticker2
    FROM ticker_list a
    JOIN ticker_list b ON a.ticker < b.ticker
),
pair_corr AS (
    SELECT
        p.ticker1,
        p.ticker2,
        CORR(r1.daily_return, r2.daily_return) AS correlation
    FROM pairs p
    JOIN returns_filtered r1 ON r1.ticker = p.ticker1
    JOIN returns_filtered r2 ON r2.ticker = p.ticker2 AND r2.date = r1.date
    GROUP BY p.ticker1, p.ticker2
    HAVING COUNT(*) >= 30
)
SELECT
    ticker1 || ' / ' || ticker2 AS pair_label,
    correlation
FROM pair_corr
ORDER BY correlation ASC
LIMIT 10

Identify the 10 stock pairs with the highest pairwise Pearson correlation of daily returns (excluding SPY), for display as a horizontal bar chart labeled 'TICKER1 / TICKER2'.

WITH returns AS (
    SELECT
        date,
        ticker,
        (close - LAG(close) OVER (PARTITION BY ticker ORDER BY date)) / LAG(close) OVER (PARTITION BY ticker ORDER BY date) AS daily_return
    FROM daily_prices
    WHERE ticker != 'SPY'
),
returns_clean AS (
    SELECT date, ticker, daily_return
    FROM returns
    WHERE daily_return IS NOT NULL
),
ticker_list AS (
    SELECT DISTINCT ticker FROM returns_clean
),
pairs AS (
    SELECT
        t1.ticker AS ticker1,
        t2.ticker AS ticker2
    FROM ticker_list t1
    JOIN ticker_list t2 ON t1.ticker < t2.ticker
),
pair_returns AS (
    SELECT
        p.ticker1,
        p.ticker2,
        r1.daily_return AS ret1,
        r2.daily_return AS ret2
    FROM pairs p
    JOIN returns_clean r1 ON r1.ticker = p.ticker1
    JOIN returns_clean r2 ON r2.ticker = p.ticker2 AND r2.date = r1.date
),
correlations AS (
    SELECT
        ticker1,
        ticker2,
        CORR(ret1, ret2) AS correlation
    FROM pair_returns
    GROUP BY ticker1, ticker2
)
SELECT
    ticker1 || ' / ' || ticker2 AS pair_label,
    correlation
FROM correlations
ORDER BY correlation DESC
LIMIT 10
